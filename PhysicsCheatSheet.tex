    \section{Kinematics}
    Position $\ne$ Displacement

    $\vec{d_1}=20m[E]$

    $\Delta \vec{d}=\vec{d}_{final}-\vec{d}_{initial}$

    $\Delta \vec{d}_{total}=\Delta\vec{d}_{1}+\Delta\vec{d}_{2}$

    $\Delta \vec{d}=\vec{v}\times \Delta t$

    $\vec{a}=\frac{\Delta \vec{v}}{\Delta \vec{t}}=\frac{\vec{v}_f-\vec{v}_i}{t_2-t_1}$

    $\vec{v}_f=\vec{v}_i+\vec{a}\times \Delta t$

    % \section{Comparing Graphs of Linear Motion}
    \section{Five Key Equations}
    \begin{align*}
        \vec{a}&=\frac{\vec{v}_f-\vec{v}_i}{\Delta t} & \Delta \vec{d}\\
        \Delta \vec{d}&=\frac{\vec{v}_f+\vec{v}_i}{2}\Delta t&\vec{a}\\
        \Delta \vec{d}&=\vec{v}_1\Delta t+\frac{1}{2}\vec{a}\Delta t^2&\vec{v}_2 \\
        \Delta \vec{d}&=\vec{v}_2\Delta t-\frac{1}{2}\vec{a}\Delta t^2&\vec{v}_1 \\
        \vec{v}_2^2&=\vec{v}_1^2+2\vec{a}\Delta \vec{d}&\Delta t\\
    \end{align*}
