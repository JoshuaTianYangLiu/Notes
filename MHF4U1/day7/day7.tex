$y=a\sin(k(x-d))+c$

\begin{itemize}
    \item amplitude $a = \frac{\text{max}-\text{min}}{2}$
    \item $k=\frac{2\pi}{\text{period}}$
    \item period $=\frac{2\pi}{k}$
    \item $d=\text{phase shift}$
    \item axis $=c=\frac{\text{max}+\text{min}}{2}$
\end{itemize}
